<script>

    function calcularResultado() {

        let total = 0;

        let perguntas = document.querySelectorAll("input[type='radio']:checked");


        if (perguntas.length < 2) { // Garante que todas as perguntas foram respondidas

            document.getElementById("result").innerHTML = "⚠️ Responda todas as perguntas antes de continuar.";

            return;

        }


        perguntas.forEach((resposta) => {

            total += parseInt(resposta.value);

        });


        let resultadoTexto = "";

        

        if (total <= 4) {

            resultadoTexto = "🔴 Nível Iniciante - A governança precisa ser estruturada!";

        } else if (total <= 7) {

            resultadoTexto = "🟡 Nível Básico - Há iniciativas, mas precisam de amadurecimento.";

        } else if (total <= 10) {

            resultadoTexto = "🟠 Nível Intermediário - A empresa está avançando, mas há pontos de melhoria.";

        } else if (total <= 14) {

            resultadoTexto = "🟢 Nível Avançado - Ótimas práticas, mas sempre há espaço para evolução!";

        } else {

            resultadoTexto = "🏆 Nível Excelente - A empresa segue as melhores práticas de governança!";

        }


        document.getElementById("result").innerHTML = resultadoTexto;

    }

</script>
